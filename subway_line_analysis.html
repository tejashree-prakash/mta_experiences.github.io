# Distribution of Delay Frequency Responses 
# Interactive Plotly Stacked Bar Chart 

dist_delay <- scored_data |>
  filter(!is.na(frequency_of_delays)) |>
  count(subway_line, group_color, frequency_of_delays, name = "n") |>
  group_by(subway_line, group_color) |>
  mutate(prop = n / sum(n)) |>
  ungroup() |>
  mutate(
    subway_line = factor(subway_line, levels = unique(subway_line))
  )

plot_ly(
  data = dist_delay,
  x = ~subway_line,
  y = ~prop,
  type = "bar",
  color = ~frequency_of_delays,
  colors = viridisLite::viridis(5),
  customdata = ~group_color,
  hovertemplate = paste(
    "Subway Line: %{x}<br>",
    "Line Group: %{customdata}<br>",
    "Proportion of Riders: %{y:.1%}<br>",
    "<extra></extra>"
  )
) |>
  layout(
    barmode = "stack",
    title = list(
      text = "Distribution of Delay Frequency Responses by Subway Line",
      x = 0.5
    ),
    xaxis = list(title = "Subway Line"),
    yaxis = list(title = "Proportion", tickformat = ".0%"),
    showlegend = TRUE
  )
# Ordinal frequency of delay responses
# Interactive plotly heat map


keep_delays <- c(
  "Rarely",
  "A few times a week",
  "Everyday"
)

heat_data <- scored_data |>
  filter(!is.na(frequency_of_delays)) |>
  filter(frequency_of_delays %in% keep_delays) |>
  count(subway_line, group_color, frequency_of_delays, name = "n") |>
  complete(
    subway_line = unique(subway_directory$subway_line),
    frequency_of_delays = keep_delays,
    fill = list(n = 0)
  ) |>
  group_by(subway_line) |>
  mutate(prop = n / sum(n)) |>
  ungroup() |>
  left_join(subway_directory, by = "subway_line") |>
  mutate(
    subway_line = factor(subway_line, levels = unique(subway_line)),
    frequency_of_delays = factor(frequency_of_delays, levels = keep_delays)
  )

plot_ly(
  data = heat_data,
  x = ~frequency_of_delays,
  y = ~subway_line,
  z = ~prop,
  type = "heatmap",
  colors = viridisLite::viridis(100),
  customdata = ~n,
  hovertemplate = paste(
    "Line: %{y}<br>",
    "Delay Frequency: %{x}<br>",
    "Proportion: %{z:.1%}<br>",
    "Count: %{customdata}<br>",
    "<extra></extra>"
  )
) |>
  layout(
    title = "Delay Frequency Distribution by Subway Line",
    xaxis = list(title = "Delay Frequency"),
    yaxis = list(title = "Subway Line"),
    margin = list(l = 90, r = 20, b = 80, t = 80)
  )

This heatmap highlights system-wide patterns, showing consistently high weekly delay reports and notable everyday delays on lines like the 7. Hover details reveal the precise counts and proportions underlying each cell.

# To establish proportion of Riders with Long Delays (â‰¥20 min): Define long delay buckets.

# Proportion of Riders with Long Delays (â‰¥20 min)
# Interactive plotly bar chart

long_categories <- c(
  "20 - 40 min", "40 - 60 min", "60 - 90 min",
  "90 min - 2 hours", "> 60 min"
)

line_order <- subway_directory |>
  arrange(group_color, subway_line) |>
  pull(subway_line)

long_delay_by_line <- scored_data |>
  filter(!is.na(average_length_subway_ride)) |>
  mutate(long_delay = average_length_subway_ride %in% long_categories) |>
  count(subway_line, group_color, hex_color, long_delay, name = "n") |>
  group_by(subway_line, group_color, hex_color) |>
  mutate(prop = n / sum(n)) |>
  filter(long_delay) |>
  ungroup() |>
  mutate(
    subway_line = factor(subway_line, levels = line_order)
  )

plot_ly(
  data = long_delay_by_line,
  x = ~subway_line,
  y = ~prop,
  type = "bar",
  marker = list(color = ~hex_color),
  customdata = ~group_color,
  hovertemplate = paste(
    "<b>Subway Line:</b> %{x}<br>",
    "<b>Long Delay Proportion:</b> %{y:.2f}<br>",
    "<b>Line Group:</b> %{customdata}<br>",
    "<extra></extra>"
  )
) |>
  layout(
    title = "Proportion of Riders Reporting Long Delays (â‰¥20 min)",
    xaxis = list(title = "Subway Line", categoryarray = line_order),
    yaxis = list(title = "Proportion", tickformat = ".0%"),
    bargap = 0.20
  )

This bar chart shows the share of riders on each line who report experiencing long delays of 20 minutes or longer. While most lines cluster between 10-20%, the Z line stands out with the highest proportion, indicating substantially more severe delays among its riders. Interactive hover labels display each lineâ€™s exact proportion and color group for more detailed comparison.
