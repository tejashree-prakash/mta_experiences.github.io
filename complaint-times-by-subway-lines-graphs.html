Step 1: Extract time of day (hour bins)

rider_data_full <- rider_data_full %>%
  # Convert submitted_at to datetime
  mutate(
    submitted_at = mdy_hm(submitted_at),  # month/day/year hour:minute (AM/PM)
    hour = hour(submitted_at),
    time_of_day = case_when(
      hour >= 6 & hour < 12 ~ "Morning",
      hour >= 12 & hour < 17 ~ "Afternoon",
      hour >= 17 & hour < 21 ~ "Evening",
      TRUE ~ "Night"
    ),
    time_of_day = factor(time_of_day, levels = c("Morning", "Afternoon", "Evening", "Night"))
  )

Step 2: Aggregate data for plotting

complaints_by_time <- rider_data_full %>%
  group_by(subway_line_used_most_often, time_of_day, complaint) %>%
  summarise(count = n(), .groups = "drop")

Step 3: Create interactive stacked bar plot with ggplot2 + plotly

p <- ggplot(complaints_by_time, 
            aes(x = time_of_day, y = count, fill = complaint, text = paste("Complaint:", complaint, "<br>Count:", count))) +
  geom_bar(stat = "identity", position = "stack") +
