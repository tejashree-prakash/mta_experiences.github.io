subway_directory_display_final =
  subway_directory_display_final %>%
  arrange(rank) %>%
  mutate(
    rank = dplyr::case_when(
      rank == 1 ~ "ðŸ¥‡ 1",
      rank == 2 ~ "ðŸ¥ˆ 2",
      rank == 3 ~ "ðŸ¥‰ 3",
      TRUE      ~ as.character(rank)
    )
  ) %>%
  gt() %>%
  tab_header(title = "NYC Subway Line Directory with Composite Score") %>%
  cols_move(
    columns = c(composite_score, rank),
    after = Color
  ) %>%
  cols_label(
    composite_score = "Score",
    rank            = "Rank"
  ) %>%
  fmt_number(
    columns = composite_score,
    decimals = 2
  ) %>%
  text_transform(
    locations = cells_body(columns = Color),
    fn = function(x) {
      mapply(function(col) {
        html(
          paste0(
            "<div style='background-color:", col,
            "; width: 60px; height: 20px; border-radius: 4px'></div>"
          )
        )
      }, x)
    }
  ) %>%
  cols_width(
    `Line Family`   ~ px(120),
    Lines           ~ px(100),
    Color           ~ px(70),
    composite_score ~ px(80),
    rank            ~ px(70)
  ) %>%
  cols_align(
    align = "center",
    columns = rank
  ) %>%
  opt_table_outline()
